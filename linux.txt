CREATE TABLE pracownik(id_pracownika INT PRIMARY KEY,
    -> imie VARCHAR(40) NOT NULL,
    -> nazwisko VARCHAR(50) NOT NULL,
    -> zarobki DOUBLE(10,2) DEFAULT 4000,
    -> data_urodzenia DATE NOT NULL,
    -> stanowisko ENUM('manager','kierownik','sprzedawca','kasjer'));
